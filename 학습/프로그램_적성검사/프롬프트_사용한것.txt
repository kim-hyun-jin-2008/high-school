
console.log("\n=== 예제 6: 할인율 계산 ===");
let price = 50000;
let discount = 0;
if (price >= 100000) {
    discount = 20;
    console.log("20% 할인이 적용됩니다.");
} else if (price >= 50000) {
    discount = 10;
    console.log("10% 할인이 적용됩니다.");
} else {
    discount = 0;
    console.log("할인이 없습니다.");
}
let finalPrice = price - (price * discount / 100);
console.log("최종 가격: " + finalPrice + "원");
